PT_CCN_Start_Civil_War = {
	CHN = {
		if = {
			limit = {
				any_country_with_original_tag = {
					original_tag_to_check = CHN
					has_government = fascism
					NOT = {	has_army_manpower = { size > 150000 } }
					NOT = {	has_army_manpower = { size < 200000 } }
				}
			}
			load_oob = CHN_1007
			load_oob = CHN_1007
		}
		CCN = {
			transfer_technology = yes
			every_state = {
				if = {
					limit = { has_state_flag = PT_thorax_rebel_state }
					transfer_state_to = CCN 
				}
			}
			#transfer_state = 322
			#transfer_state = 320 
			#transfer_state = 319
			#transfer_state = 323				
			every_state = {
				limit = {
					OR = {
						state = 323
						state = 322
						state = 320
						state = 319
					}
				}
				transfer_state_to = CCN
			}
		}
		transfer_units_fraction = {
			target = CCN
			target_organization = 1
			source_organization = 1
			keep_unit_leaders_trigger = { always = yes }
			size = 0.1
			navy_ratio = 0
		}
		CCN = {
			load_oob = CCN_Civil_War
			add_ideas = CCN_Civil_War
			declare_war_on = { target = CHN type = civil_war }
		}
		CHN = {
			add_ideas = CCN_Civil_War
			CHN_vect = {
				set_nationality = CCN
			}
			CHN_azure_lore = {
				set_nationality = CCN
			}
		}
		every_state = {
			if = {
				limit = { is_core_of = CHN }
				add_core_of = CCN
			}
			if = {
				limit = { is_claimed_by = CHN }
				add_claim_by = CCN
			}
		}		
		activate_mission = PT_civil_war_escalation_0
	}
	set_global_flag = PT_changeling_civil_war
	CCN = {
		activate_mission = PT_civil_war_escalation_0
	}
	log = "[GetDateText]: Thoraxian Civil War Started!!"
}

	set_variable = { TRO_drugs_friendly_relations_status^1 = var:TRO_drugs_enemy_influence_status^1 }
