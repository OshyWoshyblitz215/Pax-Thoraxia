	CCN = {
		for_loop_effect = {
			end = 5

			break = loop_break
			if = {
				limit = {
					check_variable = {
						var = PT_queen_stage_option
						value = 0
						compare = equals
					}
				}
				set_variable = { loop_break = 1 }
			}
			else = {
				set_variable = { PT_queen_stage_option^temp = 1 }
			}
		}
		set_variable = { PT_selected_queen_action_option = 1 }
		add_to_variable = { PT_CCN_queen_dirty_bit = 0.001 }
	}
}
