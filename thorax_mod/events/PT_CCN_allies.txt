add_namespace = PT_CCN_allies


country_event = {
	id = PT_CCN_allies.1
	title = PT_CCN_allies.1.t
	desc = PT_CCN_allies.1.d
	fire_only_once = yes
	is_triggered_only = yes
	immediate = {
		if = {
			limit = {
				CCN = {
					is_in_faction = no
				}
			}
			CCN = {
				create_faction = THORAX_CIVIL_WAR_ALLIANCE
			}
		}
		VRK = {
			leave_faction = yes
			add_country_leader_trait = PT_swayed_by_thorax
			add_opinion_modifier = {
				target = CCN
				modifier = PT_they_like_thorax
			}
			add_ai_strategy = {
				type = alliance
				id = CCN
				value = 500
			}
			add_ai_strategy = {
				type = befriend
				id = CCN
				value = 500
			}
			add_ai_strategy = {
 				type = diplo_action_acceptance
				id = CCN
 				value = 100
 				target = join_allies
			}
		}
		add_opinion_modifier = {
				target = VRK
				modifier = PT_they_like_thorax
			}
		every_country = {
			limit = {
				OR = {
					is_in_faction_with = CCN
					original_tag = CCN
				}
			}
			white_peace = VRK
		}
		CCN = {
			add_to_faction = VRK
		}
	}
	option = {
		name = PT_CCN_allies.4.a
	}
}
