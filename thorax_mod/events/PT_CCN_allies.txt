add_namespace = PT_CCN_allies


country_event = {
	id = PT_CCN_allies.1
	fire_only_once = yes
	is_triggered_only = yes
	immediate = {
		if = {
			limit = {
				CCN = {
					is_in_faction = no
				}
			}
			CCN = {
				create_faction = THORAX_CIVIL_WAR_ALLIANCE
			}
		}
		VRK = { leave_faction = yes }
		every_country = {
			limit = {
				OR = {
					is_in_faction_with = CCN
					original_tag = CCN
				}
			}
			white_peace = VRK
		}
		CCN = {
			add_to_faction = VRK
		}
	}
	option = {
		name = PT_CCN_allies.4.a
	}
}
